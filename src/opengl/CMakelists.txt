set(SRC_PATH ${CMAKE_CURRENT_SOURCE_DIR}/..)
set(EXT_PATH ${SRC_PATH}/../../external)

set(HEADER_FILES 
		GraphicsContext.h
	)

set(SOURCE_FILES 
		GraphicsContext.cpp
		${EXT_PATH}/glad/src/glad.c
	)


add_library(gbcemu_graphics_library
		${SOURCE_FILES}
		${HEADER_FILES}
	)

target_include_directories(gbcemu_graphics_library PUBLIC ${SRC_PATH})
target_include_directories(gbcemu_graphics_library PUBLIC ${EXT_PATH}/glad/include)
target_include_directories(gbcemu_graphics_library PUBLIC ${EXT_PATH}/glfw/include)

target_link_libraries(gbcemu_graphics_library PRIVATE gbcemu_util)
target_link_libraries(gbcemu_graphics_library PRIVATE glad)
target_link_libraries(gbcemu_graphics_library PRIVATE glfw)
